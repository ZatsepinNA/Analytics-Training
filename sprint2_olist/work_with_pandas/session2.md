# ✅ ПРАВИЛЬНО: Сначала флаг, потом статистика, потом фильтр
df_orders['is_delivered'] = df_orders['order_delivered_customer_date'].notna()
print(df_orders['is_delivered'].value_counts())
delivered_orders = df_orders[df_orders['is_delivered'] == 1].copy()

# ✅ ПРАВИЛЬНО: Фильтрация с & (не and!)
mask = (df_orders['order_status'] == 'delivered') & (df_orders['order_year'] == 2018)
filtered = df_orders.loc[mask]

# ❌ НЕПРАВИЛЬНО: and вместо &
filtered = df_orders[df_orders['status'] == 'delivered' and df_orders['year'] == 2018]

# ❌ НЕПРАВИЛЬНО: Флаг после удаления (бесполезен)
